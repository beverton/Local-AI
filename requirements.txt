# Haupt-Requirements - installiert alle Dependencies
# Für minimale Installation siehe requirements-base.txt

# Installation:
#   pip install -r requirements.txt          # Installiert alles
#   pip install -r requirements-base.txt    # Nur Basis (Text-Modelle)
#   pip install -r requirements-base.txt -r requirements-image.txt  # Basis + Bildgenerierung
#   pip install -r requirements-base.txt -r requirements-audio.txt   # Basis + Audio

# Basis (Text-Modelle) - immer nötig
-r requirements-base.txt

# Bildgenerierung (optional - nur wenn Flux genutzt wird)
-r requirements-image.txt

# Audio (optional - nur wenn Whisper genutzt wird)
-r requirements-audio.txt
